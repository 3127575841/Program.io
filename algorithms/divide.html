归并排序


// 归并排序
#include
using namespace std;

const int N = 100010;
int a[N], tmp[N];

void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;
    
    int mid = l + r >> 1;
    merge_sort(q, l, mid);
    merge_sort(q, mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k++] = q[i++];
        else tmp[k++] = q[j++];
    
    while (i <= mid) tmp[k++] = q[i++];
    while (j <= r) tmp[k++] = q[j++];
    
    for (i = l, j = 0; i <= r; i++, j++) q[i] = tmp[j];
        }

int main()
{
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    
    merge_sort(a, 0, n - 1);
    
    for (int i = 0; i < n; i++) printf("%d ", a[i]);
    
    return 0;
}

快速排序


// 快速排序
#include
using namespace std;

const int N = 100010;
int a[N];

void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;
    
    int x = q[l + r >> 1], i = l - 1, j = r + 1;
    while (i < j)
    {
        do i++; while (q[i] < x);
        do j--; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
    }
    
int main()
{
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    
    quick_sort(a, 0, n - 1);
    
    for (int i = 0; i < n; i++) printf("%d ", a[i]);
    
    return 0;
}

二分查找


// 二分查找
#include
using namespace std;

const int N = 100010;
int a[N];

// 查找>=x的第一个位置
int lower_bound(int q[], int l, int r, int x)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (q[mid] >= x) r = mid;
        else l = mid + 1;
    }
    return l;
    }

// 查找>x的第一个位置
int upper_bound(int q[], int l, int r, int x)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (q[mid] > x) r = mid;
        else l = mid + 1;
    }
    return l;
}

int main()
{
    int n, m;
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    
    while (m--)
    {
        int x;
        scanf("%d", &x);
        int l = lower_bound(a, 0, n, x);
        int r = upper_bound(a, 0, n, x) - 1;
        
        if (l <= r) printf("%d %d\n", l, r);
        else printf("-1 -1\n");
        }
    
    return 0;
}
